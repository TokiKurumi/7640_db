-- Create Database
CREATE DATABASE IF NOT EXISTS ecommerce_platform DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE ecommerce_platform;

-- 1. Vendors Table (matches basic vendor field requirements)
DROP TABLE IF EXISTS transactions;
DROP TABLE IF EXISTS order_items;
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS products;
DROP TABLE IF EXISTS customers;
DROP TABLE IF EXISTS vendors;

CREATE TABLE vendors (
    vendor_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the vendor',
    business_name VARCHAR(100) NOT NULL COMMENT 'Business name',
    avg_rating DECIMAL(2,1) DEFAULT 0.0 COMMENT 'Average customer rating',
    geo_presence VARCHAR(200) NOT NULL COMMENT 'Geographical presence',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Vendor information table';

-- 2. Customers Table (matches basic customer field requirements)
CREATE TABLE customers (
    customer_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the customer',
    contact_number VARCHAR(20) NOT NULL UNIQUE COMMENT 'Contact phone number',
    shipping_address VARCHAR(200) NOT NULL COMMENT 'Shipping address',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Customer information table';

-- 3. Products Table (matches basic product field requirements, max 3 tags)
CREATE TABLE products (
    product_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the product',
    vendor_id INT NOT NULL COMMENT 'ID of the vendor who owns the product',
    product_name VARCHAR(100) NOT NULL COMMENT 'Product name',
    price DECIMAL(10,2) NOT NULL COMMENT 'Selling price',
    stock_quantity INT NOT NULL DEFAULT 0 COMMENT 'Stock quantity',
    tag1 VARCHAR(50) DEFAULT NULL COMMENT 'Product tag 1',
    tag2 VARCHAR(50) DEFAULT NULL COMMENT 'Product tag 2',
    tag3 VARCHAR(50) DEFAULT NULL COMMENT 'Product tag 3',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (stock_quantity >= 0),
    CHECK (price > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Product information table';

-- 4. Orders Table (matches basic order field requirements, supports cross-vendor orders)
CREATE TABLE orders (
    order_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the order',
    customer_id INT NOT NULL COMMENT 'ID of the customer who placed the order',
    order_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT 'Order creation time',
    total_price DECIMAL(10,2) NOT NULL DEFAULT 0.00 COMMENT 'Total price of the order',
    order_status ENUM('pending_payment','pending_shipment','shipped','cancelled') DEFAULT 'pending_payment' COMMENT 'Order status',
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Main order table';

-- 5. Order-Product Association Table (order_items) (solves many-to-many relationship, supports cross-vendor)
CREATE TABLE order_items (
    order_item_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the order item',
    order_id INT NOT NULL COMMENT 'ID of the order it belongs to',
    product_id INT NOT NULL COMMENT 'Product ID',
    vendor_id INT NOT NULL COMMENT 'ID of the vendor who owns the product',
    quantity INT NOT NULL COMMENT 'Purchase quantity',
    item_price DECIMAL(10,2) NOT NULL COMMENT 'Product unit price at the time of order',
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (quantity > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Order details table';

-- 6. Transactions Table (matches basic transaction field requirements)
CREATE TABLE transactions (
    transaction_id INT PRIMARY KEY AUTO_INCREMENT COMMENT 'Unique ID for the transaction',
    order_id INT NOT NULL COMMENT 'Associated order ID',
    customer_id INT NOT NULL COMMENT 'ID of the paying customer',
    vendor_id INT NOT NULL COMMENT 'ID of the receiving vendor',
    transaction_amount DECIMAL(10,2) NOT NULL COMMENT 'Transaction amount',
    transaction_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT 'Transaction time',
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id) ON DELETE CASCADE,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (transaction_amount > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Transaction records table';
