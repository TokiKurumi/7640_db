-- 创建数据库
CREATE DATABASE IF NOT EXISTS ecommerce_platform DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE ecommerce_platform;

-- 1. 供应商表 vendors (匹配基础要求的vendor字段)
DROP TABLE IF EXISTS transactions;
DROP TABLE IF EXISTS order_items;
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS products;
DROP TABLE IF EXISTS customers;
DROP TABLE IF EXISTS vendors;

CREATE TABLE vendors (
    vendor_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '供应商唯一ID',
    business_name VARCHAR(100) NOT NULL COMMENT '商家名称',
    avg_rating DECIMAL(2,1) DEFAULT 0.0 COMMENT '客户平均评分',
    geo_presence VARCHAR(200) NOT NULL COMMENT '地理位置',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='供应商信息表';

-- 2. 客户表 customers (匹配基础要求的customer字段)
CREATE TABLE customers (
    customer_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '客户唯一ID',
    contact_number VARCHAR(20) NOT NULL UNIQUE COMMENT '联系电话',
    shipping_address VARCHAR(200) NOT NULL COMMENT '收货地址',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='客户信息表';

-- 3. 产品表 products (匹配基础要求的product字段，最多3个标签)
CREATE TABLE products (
    product_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '产品唯一ID',
    vendor_id INT NOT NULL COMMENT '所属供应商ID',
    product_name VARCHAR(100) NOT NULL COMMENT '产品名称',
    price DECIMAL(10,2) NOT NULL COMMENT '售价',
    stock_quantity INT NOT NULL DEFAULT 0 COMMENT '库存数量',
    tag1 VARCHAR(50) DEFAULT NULL COMMENT '产品标签1',
    tag2 VARCHAR(50) DEFAULT NULL COMMENT '产品标签2',
    tag3 VARCHAR(50) DEFAULT NULL COMMENT '产品标签3',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (stock_quantity >= 0),
    CHECK (price > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='产品信息表';

-- 4. 订单表 orders (匹配基础要求的order字段，支持跨供应商订单)
CREATE TABLE orders (
    order_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '订单唯一ID',
    customer_id INT NOT NULL COMMENT '下单客户ID',
    order_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '订单创建时间',
    total_price DECIMAL(10,2) NOT NULL DEFAULT 0.00 COMMENT '订单总价',
    order_status ENUM('pending_payment','pending_shipment','shipped','cancelled') DEFAULT 'pending_payment' COMMENT '订单状态',
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单主表';

-- 5. 订单-产品关联表 order_items (解决订单与产品多对多关系，支持跨供应商)
CREATE TABLE order_items (
    order_item_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '订单项唯一ID',
    order_id INT NOT NULL COMMENT '所属订单ID',
    product_id INT NOT NULL COMMENT '产品ID',
    vendor_id INT NOT NULL COMMENT '产品所属供应商ID',
    quantity INT NOT NULL COMMENT '购买数量',
    item_price DECIMAL(10,2) NOT NULL COMMENT '下单时的产品单价',
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (quantity > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单详情表';

-- 6. 交易表 transactions (匹配基础要求的transaction字段)
CREATE TABLE transactions (
    transaction_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '交易唯一ID',
    order_id INT NOT NULL COMMENT '关联订单ID',
    customer_id INT NOT NULL COMMENT '付款客户ID',
    vendor_id INT NOT NULL COMMENT '收款供应商ID',
    transaction_amount DECIMAL(10,2) NOT NULL COMMENT '交易金额',
    transaction_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '交易时间',
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id) ON DELETE CASCADE,
    FOREIGN KEY (vendor_id) REFERENCES vendors(vendor_id) ON DELETE CASCADE,
    CHECK (transaction_amount > 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='交易记录表';
